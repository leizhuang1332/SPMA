数据库表设计（新增 3 张表）

-- 文件路径缓存——供 repo_router 快速定位
CREATE TABLE file_path_cache (
    id SERIAL PRIMARY KEY,
    repo_name VARCHAR(255) NOT NULL,
    file_path TEXT NOT NULL,
    file_type VARCHAR(50),        -- py, java, go, ts, ...
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(repo_name, file_path)
);
CREATE INDEX idx_fpc_file_path ON file_path_cache USING gin(file_path gin_trgm_ops);

-- 代码元数据——AST 调用图
CREATE TABLE code_metadata (
    id SERIAL PRIMARY KEY,
    repo_name VARCHAR(255) NOT NULL,
    file_path TEXT NOT NULL,
    function_name VARCHAR(255),
    calls JSONB DEFAULT '[]',      -- [{"func": "x", "file": "y.py"}]
    called_by JSONB DEFAULT '[]',
    imports JSONB DEFAULT '[]',
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(repo_name, file_path, function_name)
);

-- 仓库注册表——目录→模块映射
CREATE TABLE repo_registry (
    id SERIAL PRIMARY KEY,
    repo_name VARCHAR(255) NOT NULL UNIQUE,
    repo_url TEXT NOT NULL,
    local_path TEXT NOT NULL,
    dir_module_map JSONB DEFAULT '{}', -- {"src/auth/": "认证模块"}
    languages JSONB DEFAULT '[]',      -- ["python", "typescript"]
    last_indexed_at TIMESTAMPTZ,
    enabled BOOLEAN DEFAULT true
);

Webhook 增量更新流程

1. Git push → Webhook POST 到达
2. git_manager.handle_webhook() 解析 payload，提取 repo_name + changed_files
3. 10s 防抖（合并同一 push 的多次 webhook）
4. git pull 仓库到最新
5. file_path_cache.incremental_update(changed_files) → 增量更新路径缓存
6. ast_parser.parse_files(changed_files) → 增量更新调用图
7. gitlog_req_extractor.extract_new_links() → 提取新 commit 的 REQ 关联
8. 写入 search_logger: "code ingestion: {repo} +{n} files"
