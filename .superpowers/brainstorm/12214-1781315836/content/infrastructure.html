Postgres 冷 Trace 存储 + Agent Dashboard + 告警

1. Postgres 冷 Trace 存储

参照 SearchLogger 的 async queue + DB write 模式。查询结束后异步写入，不阻塞 Agent 循环。

agent_traces 表 + agent_rounds 表
-- 查询级 trace（一次用户查询对应一条）
CREATE TABLE agent_traces (
    query_id UUID PRIMARY KEY,
    session_id VARCHAR(255),
    original_query TEXT NOT NULL,
    classification JSONB,            -- ClassificationResult
    entities JSONB,                  -- ExtractedEntities
    rewritten_queries JSONB,         -- {source: rewritten_query}
    worker_outputs JSONB,            -- WorkerOutput[]
    quality_scores JSONB,            -- {worker_type: score}
    reschedule_count INT DEFAULT 0,
    final_answer TEXT,
    synthesis_output JSONB,          -- SynthesisOutput
    total_latency_ms INT,
    total_llm_calls INT,
    total_tokens INT,
    convergence_reason VARCHAR(255),
    degradation_level VARCHAR(10),   -- L0-L5
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 轮次级 trace（每个 Agent 的每轮状态快照）
CREATE TABLE agent_rounds (
    id SERIAL PRIMARY KEY,
    query_id UUID REFERENCES agent_traces(query_id),
    agent_type VARCHAR(50),          -- supervisor/doc/code/sql/synthesis
    round_num INT,
    action VARCHAR(100),             -- search/assess/expand/generate/audit
    results_summary JSONB,           -- 本轮结果摘要（截断 ≤2KB）
    assessment VARCHAR(50),
    confidence FLOAT,
    latency_ms INT,
    llm_calls INT,
    tokens_used INT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_agent_rounds_query ON agent_rounds(query_id, agent_type, round_num);
CREATE INDEX idx_agent_traces_created ON agent_traces(created_at);

TraceLogger 设计

# src/spma/observability/trace_logger.py
class AgentTraceLogger:  # 参照 SearchLogger 模式
    def __init__(self, db_pool, max_queue=500)
    async def start()              # 启动后台 worker
    async def log_query(query_id, state_snapshot)  # 查询结束后写入
    async def log_round(query_id, agent, round_num, snapshot)  # 每轮写入
    async def get_trace(query_id) → AgentTrace  # 读取完整 trace
    async def stop()               # 排空队列后停止

# 写入时机: Supervisor 图.ainvoke() 返回后 + Synthesis 图返回后
# 写入策略: Write-back（异步，不阻塞 Agent 循环）
# 延迟目标: 查询结束后 < 5s 内 trace 落库
